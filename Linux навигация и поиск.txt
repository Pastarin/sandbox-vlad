1) cd /html - сменяем каталог с низпадающим поиском нужного наименования каталога
2) ls -d */ - вывод каталогов в текущей директории
3) cat error.txt | grep -Ev "^$" > out.txt - удаляем пустые строки в файле с выводом результата в новый файл out.txt
4) find /var/www/images -iname '*.jpg' - выводим все файлы с искомым расширением в конкретной директории
5) rename 's/.txt/.log/logs' *.sh - переименовуем все файлы в папке logs с .txt на .log
6) chmod 660 /images - группа и пользователь будут иметь доступ к чтении и записи в директории, остальные не будут иметь доступа вовсе
7) cat /dev/null > /var/www/logs/access.txt - очищаем содержимое файла
8) find /var/www/images -size +2M
